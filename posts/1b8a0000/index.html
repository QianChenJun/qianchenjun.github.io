    private void loadDi() {
        // 1. éå†å®¹å™¨é›†åˆ
        for (Map.Entry<Class, Object> entry : beanFactory.entrySet()) {
            // 2. å–å‡ºæ¯ä¸€ä¸ªvalue

                    field.setAccessible(true);

                    // 5. ç»™è¯¥å˜é‡èµ‹å€¼
                    try {
                        field.set(obj, beanFactory.get(field.getType()));

            }
        }
    }

    /**

    private void loadBean(File file) {

        if (file.isDirectory()) {
            // 2.å¦‚æœæ˜¯æ–‡ä»¶å¤¹ï¼Œè·å–å­æ–‡ä»¶åˆ—è¡¨
            File[] childrenFiles = file.listFiles();

            // 3.åˆ¤æ–­å­æ–‡ä»¶åˆ—è¡¨æ˜¯å¦ä¸ºç©º
            if (childrenFiles == null || childrenFiles.length == 0) {

            for (File childrenFile : childrenFiles) {

                if (childrenFile.isDirectory()) {

                // 5.3 åˆ¤æ–­å½“å‰æ–‡ä»¶æ˜¯å¦æ˜¯.classç±»å‹
                if (pathWithClass.contains(".class")) {

                    String fullName = pathWithClass
                            .replaceAll("\\\\", ".")
                            .replace(".class", "");

                    // 5.5 åˆ¤æ–­ç±»ä¸Šé¢æ˜¯å¦æœ‰@Beanæ³¨è§£ï¼Œå¦‚æœæœ‰åˆ™å®ä¾‹åŒ–
                    try {

                        if (!clazz.isInterface()) {

                            if (bean != null) {

                                // 5.6 å°†å¯¹è±¡å®ä¾‹åŒ–åæ”¾åˆ°mapé›†åˆä¸­
                                // åˆ¤æ–­ä¸€ä¸‹å½“å‰å®ä¾‹æœ‰æ²¡æœ‰æ¥å£ï¼Œæœ‰åˆ™ä½œä¸ºkey
                                if (clazz.getInterfaces().length > 0) {

                                } else {
                                    beanFactory.put(clazz, instance);
                                }

                            }
                        }
                    } catch (Exception e) {
                        throw new RuntimeException(e);
                    }
                }
            }
        }
    }
